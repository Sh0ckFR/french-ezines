le header des pcx

 

voici l'entete d'un PCX :

struct entetePCX{
char manu;
char version;
char encoding;
char bitstopixel;
int x, y;
int w, h;
int horizontalres;
int verticalres;
char ega_palette[48];
char reserved;
char colplanes;
int bytetoline;
int palette;
char buff[58];
};

pour l'utiliser :

PCXHeaderStruct PCXHeader;
unsigned int handle, taille_de_image
handle = open("fichier.pcx", O_RDONLY | O_BINARY);
read(handle,&PCXHeader,128);
taille_de_image = PCXHeader.h*PCXHeader.w;
close(handle);
